function link_scroll() {
    setTimeout(function(){
        var anchor = window.location.hash
        if(anchor!==''){
            var scroll = document.querySelector(anchor).offsetTop - 100;
            scrollbar.scrollTo(0, scroll, 1200);
        }
    },10)
}
