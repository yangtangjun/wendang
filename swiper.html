var mySwiper = new Swiper ('.swiper', {
    loop: true,
    slidesPerView : 3,
    spaceBetween: 40,
    speed: 6000,//匀速时间
    allowTouchMove: false,
    autoplay: {
        delay: 0,
        stopOnLastSlide: false,
        disableOnInteraction: false,
    },
});
// 鼠标移入暂停轮播 , 移出继续轮播
var lastNeedSwiperSpeed;
$('.notable').on('mouseenter', function(){
    lastNeedSwiperSpeed = Math.abs(Math.abs(mySwiper.getTranslate()) - Math.abs(mySwiper.translate)) / (mySwiper.slideWidth + mySwiper.params.spaceBetween) * mySwiper.params.speed;
    mySwiper.setTranslate(mySwiper.getTranslate());
    mySwiper.autoplay.stop();
})
$('.notable').on('mouseleave', function(){
    mySwiper.slideTo(mySwiper.activeIndex, lastNeedSwiperSpeed);
    mySwiper.autoplay.start();
})

JS 代码:
var interleaveOffset = 0.5; //视差比值
var mySwiper = new Swiper ('.swiper', {
    speed: 1000,
    grabCursor: true,
    watchSlidesProgress: true,
    mousewheelControl: true,
    keyboardControl: true,
    loop: true,
    autoplay: {
        delay: 2000,
        disableOnInteraction: false,
    },
    on: {
        progress: function(swiper) {
            for (var i = 0; i < swiper.slides.length; i++) {
                var slideProgress = swiper.slides[i].progress;
                var innerOffset = swiper.width * interleaveOffset;
                var innerTranslate = slideProgress * innerOffset;
                swiper.slides[i].querySelector(".slide-inner").style.transform = "translate3d(" + innerTranslate + "px, 0, 0)";
            }
        },
        touchStart: function(swiper) {
            for (var i = 0; i < swiper.slides.length; i++) {
                swiper.slides[i].style.transition = "";
            }
        },
        setTransition: function(swiper, speed) {
            for (var i = 0; i < swiper.slides.length; i++) {
                swiper.slides[i].style.transition = speed + "ms";
                swiper.slides[i].querySelector(".slide-inner").style.transition = speed + "ms";
            }
        }
    }
})
